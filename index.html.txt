let sheetData = [];
let journal = [];
let ledger = {};

const accountKeywords = {
  "Petty Cash": ["petty cash"],
  "Bank": ["bank", "cheque", "transfer"],
  "Staff Food": ["food", "canteen", "lunch", "snack"],
  "Legal": ["law", "legal", "attorney", "court"],
  "Stationery": ["stationery", "pen", "paper", "office supply"],
  "Printing": ["print", "brochure", "flyer", "poster"],
  "Car Fix": ["car", "fuel", "repair", "garage"],
  "Sales": ["sale", "sold", "invoice"],
  "Sales Returns": ["return", "sales return"],
  "Income": ["income", "revenue", "earning"],
  "Discount Received": ["discount received", "discount"],
  "Asset": ["asset", "equipment", "furniture", "computer", "vehicle"],
  "Liability": ["loan", "liability", "payable"],
  "Capital": ["capital", "owner"],
  "Depreciation": ["depreciation"],
  "Losses": ["lost", "damaged", "write-off", "obsolete"]
};

function detectAccount(description){
  if(!description) return "Unknown";
  description = description.toLowerCase();
  for(const [account, keywords] of Object.entries(accountKeywords)){
    for(const kw of keywords){
      if(description.includes(kw)) return account;
    }
  }
  return "Misc";
}

function loadExcel() {
  const file = document.getElementById('fileInput').files[0];
  if (!file) { alert("Please select an Excel file first."); return; }
  const reader = new FileReader();
  reader.onload = function (e) {
    const data = new Uint8Array(e.target.result);
    const workbook = XLSX.read(data, { type: 'array' });
    const ws = workbook.Sheets[workbook.SheetNames[0]];
    sheetData = XLSX.utils.sheet_to_json(ws);
    alert('Loaded ' + sheetData.length + ' transactions.');
  };
  reader.readAsArrayBuffer(file);
}

function generateJournal() {
  if (!sheetData.length) return alert("No data loaded!");
  const table = document.getElementById('journalTable');
  table.innerHTML = '<tr><th>Date</th><th>Account</th><th>Debit</th><th>Credit</th><th>Description</th></tr>';
  journal = [];
  const counterAccount = document.getElementById('counterAccount').value || "Petty Cash";

  sheetData.forEach(row => {
    const { Date, Description, Type, Amount, Account } = row;
    const amt = parseFloat(Amount);
    if(!Type || !amt) return;

    const accountToUse = Account || detectAccount(Description);

    // Determine debit and credit based on accounting principles
    if(Type.toLowerCase() === 'receipt'){ // Money coming in
      journal.push({ Date, Account: counterAccount, Debit: amt, Credit: '', Description });
      journal.push({ Date, Account: accountToUse, Debit: '', Credit: amt, Description });
    } else { // Payment/expense
      journal.push({ Date, Account: accountToUse, Debit: amt, Credit: '', Description });
      journal.push({ Date, Account: counterAccount, Debit: '', Credit: amt, Description });
    }
  });

  renderRowsInChunks(journal, table);
}

function generateLedger() {
  if (!journal.length) return alert("Generate journal entries first!");
  ledger = {};
  const table = document.getElementById('ledgerTable');
  table.innerHTML = '<tr><th>Account</th><th>Date</th><th>Description</th><th>Debit</th><th>Credit</th><th>Balance</th></tr>';

  journal.forEach(jr => {
    if(!ledger[jr.Account]) ledger[jr.Account] = [];
    ledger[jr.Account].push(jr);
  });

  for(const account of Object.keys(ledger)){
    let balance = 0;
    renderRowsInChunks(ledger[account], table, 500, (jr) => {
      let debit = parseFloat(jr.Debit) || 0;
      let credit = parseFloat(jr.Credit) || 0;
      balance += debit - credit;
      return `<td>${account}</td><td>${jr.Date||''}</td><td>${jr.Description||''}</td><td>${debit||''}</td><td>${credit||''}</td><td>${balance.toFixed(2)}</td>`;
    });
  }
}

function renderRowsInChunks(rows, table, chunkSize = 500, formatRow) {
  let i = 0;
  function renderChunk() {
    const fragment = document.createDocumentFragment();
    for(let j=0; j<chunkSize && i<rows.length; j++, i++){
      const jr = rows[i];
      const tr = document.createElement('tr');
      tr.innerHTML = formatRow ? formatRow(jr) : `<td>${jr.Date||''}</td><td>${jr.Account}</td><td>${jr.Debit||''}</td><td>${jr.Credit||''}</td><td>${jr.Description||''}</td>`;
      fragment.appendChild(tr);
    }
    table.appendChild(fragment);
    if(i<rows.length) setTimeout(renderChunk, 0);
  }
  renderChunk();
}

function downloadJournalExcel() {
  if(!journal.length) return alert("Generate journal first!");
  const ws = XLSX.utils.json_to_sheet(journal);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'Journal');
  XLSX.writeFile(wb, 'Journal.xlsx');
}

function downloadLedgerExcel() {
  if(!Object.keys(ledger).length) return alert("Generate ledger first!");
  let ledgerRows = [];
  for(const account in ledger){
    let balance = 0;
    ledger[account].forEach(jr => {
      let debit = parseFloat(jr.Debit) || 0;
      let credit = parseFloat(jr.Credit) || 0;
      balance += debit - credit;
      ledgerRows.push({Account: account, Date: jr.Date, Description: jr.Description, Debit: debit, Credit: credit, Balance: balance.toFixed(2)});
    });
  }
  const ws = XLSX.utils.json_to_sheet(ledgerRows);
  const wb = XLSX.utils.book_new();
  XLSX.utils.book_append_sheet(wb, ws, 'Ledger');
  XLSX.writeFile(wb, 'Ledger.xlsx');
}
